function saveToFile(filename, content) {
  fs.writeFile(path.join(__dirname, filename), content, (err) => {
    if (err) {
      console.error(`Failed to save ${filename}: ${err}`);
    } else {
      console.log(`${filename} was saved successfully.`);
    }
  });
}
