const exhibitsList = [

    description: "The original agreement outlining the ownership, control, and management structure of Crosscheck Studios, LLC."
  },
  {
    exhibitTitle: "Email Correspondence",

  },
  {

    description: "Documentation of the plaintiff's health issues and their impact on his ability to perform his duties."

  {
    exhibitTitle: "Bar Complaint Against Jeremiah Reynolds, Esq.",
    description: "A complaint filed by the plaintiff alleging unethical conduct by defendant Reynolds."
  }
];

function generateAppendicesOrExhibitsFile() {
  const filePath = path.join(__dirname, 'Appendices_or_Exhibits.txt');
  let content = `### APPENDICES OR EXHIBITS\n\n`;

  exhibitsList.forEach((exhibit, index) => {
    content += `Exhibit ${index + 1}: ${exhibit.exhibitTitle}\nDescription: ${exhibit.description}\n\n`;
  });

  fs.writeFile(filePath, content, (err) => {
    if (err) {
      console.error('Error writing Appendices or Exhibits to file:', err);
    } else {
      console.log('Appendices or Exhibits successfully written to file.');
    }
  });
}
