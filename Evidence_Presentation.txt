function generateEvidencePresentation(evidenceList) {
  let evidencePresentationContent = "### EVIDENCE PRESENTATION\n\n";
  evidenceList.forEach(evidence => {
    evidencePresentationContent += `**${evidence.exhibitNumber}**: ${evidence.description}\n\n`;
  });
  return evidencePresentationContent;
}

function saveToFile(fileName, content) {
  fs.writeFile(path.join(__dirname, fileName), content, (err) => {
    if (err) {
      console.error(`Failed to save ${fileName}: ${err}`);
    } else {
      console.log(`${fileName} has been saved.`);
    }
  });
}

const evidencePresentationContent = generateEvidencePresentation(evidenceList);
